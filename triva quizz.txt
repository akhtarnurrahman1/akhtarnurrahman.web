                <div id="answers-container" class="flex flex-col space-y-2">
                    <!-- Tombol jawaban akan dibuat oleh JavaScript -->
                </div>
                <div class="flex justify-between items-center mt-4">
                    <p id="score-text" class="text-lg font-bold">Skor: 0 / 0</p>
                    <button id="next-button" class="btn btn-answer hidden">Lanjut</button>
                    <button id="restart-button" class="btn btn-red hidden">Mulai Lagi</button>

    <script>
        // ==========================================================
        // Logika untuk Game Kuis Trivia
        // ==========================================================
        const questionText = document.getElementById('question-text');
        const answersContainer = document.getElementById('answers-container');
        const nextButton = document.getElementById('next-button');
        const restartButton = document.getElementById('restart-button');
        const scoreText = document.getElementById('score-text');

        // Menambahkan lebih banyak pertanyaan ke dalam array
        const questions = [
            {
                question: "Apa ibu kota Indonesia?",
                answers: ["Jakarta", "Surabaya", "Bandung"],
                correctAnswer: "Jakarta"
            },
            {
                question: "Planet terdekat dari Matahari adalah?",
                answers: ["Bumi", "Venus", "Merkurius", "Mars"],
                correctAnswer: "Merkurius"
            },
            {
                question: "Berapa jumlah hari dalam satu tahun kabisat?",
                answers: ["365", "366", "367"],
                correctAnswer: "366"
            },
            {
                question: "Siapakah penemu bola lampu?",
                answers: ["Thomas Edison", "Nikola Tesla", "Alexander Graham Bell"],
                correctAnswer: "Thomas Edison"
            },
            {
                question: "Alat musik tradisional Jawa yang dimainkan dengan cara dipetik adalah?",
                answers: ["Gamelan", "Angklung", "Siter", "Sasando"],
                correctAnswer: "Siter"
            },
            {
                question: "Apakah nama sungai terpanjang di dunia?",
                answers: ["Sungai Nil", "Sungai Amazon", "Sungai Yangtze"],
                correctAnswer: "Sungai Nil"
            },
            {
                question: "Berapa banyak benua yang ada di Bumi?",
                answers: ["5", "6", "7", "8"],
                correctAnswer: "7"
            },
            {
                question: "Cabang ilmu biologi yang mempelajari tentang serangga adalah?",
                answers: ["Botani", "Zoologi", "Entomologi", "Mikrobiologi"],
                correctAnswer: "Entomologi"
            },
            {
                question: "Apakah nama film animasi yang diproduksi oleh Pixar tentang mainan yang hidup?",
                answers: ["Finding Nemo", "Toy Story", "Up"],
                correctAnswer: "Toy Story"
            },
            {
                question: "Siapakah pendiri Microsoft?",
                answers: ["Steve Jobs", "Bill Gates", "Mark Zuckerberg"],
                correctAnswer: "Bill Gates"
            }
        ];

        let currentQuestionIndex = 0;
        let score = 0;

        function showQuestion() {
            const currentQuestion = questions[currentQuestionIndex];
            questionText.textContent = currentQuestion.question;
            answersContainer.innerHTML = ''; // Hapus tombol jawaban sebelumnya
            nextButton.classList.add('hidden');

            currentQuestion.answers.forEach(answer => {
                const button = document.createElement('button');
                button.textContent = answer;
                button.classList.add('btn', 'btn-answer');
                button.addEventListener('click', () => selectAnswer(answer));
                answersContainer.appendChild(button);
            });
        }

        function selectAnswer(selectedAnswer) {
            const currentQuestion = questions[currentQuestionIndex];
            const isCorrect = selectedAnswer === currentQuestion.correctAnswer;

            // Nonaktifkan semua tombol setelah jawaban dipilih
            answersContainer.querySelectorAll('.btn-answer').forEach(btn => {
                btn.disabled = true;
                if (btn.textContent === currentQuestion.correctAnswer) {
                    btn.classList.add('btn-correct');
                } else if (btn.textContent === selectedAnswer) {
                    btn.classList.add('btn-wrong');
                }
            });

            if (isCorrect) {
                score++;
            }

            updateScore();
            nextButton.classList.remove('hidden');
        }

        function updateScore() {
            scoreText.textContent = `Skor: ${score} / ${currentQuestionIndex + 1}`;
        }

        function nextQuestion() {
            currentQuestionIndex++;
            if (currentQuestionIndex < questions.length) {
                showQuestion();
            } else {
                showResult();
            }
        }

        function showResult() {
            questionText.textContent = `Kuis selesai! Skor akhir Anda: ${score} dari ${questions.length}.`;
            answersContainer.innerHTML = '';
            nextButton.classList.add('hidden');
            restartButton.classList.remove('hidden');
        }

        function restartQuiz() {
            currentQuestionIndex = 0;
            score = 0;
            restartButton.classList.add('hidden');
            updateScore();
            showQuestion();
        }

        nextButton.addEventListener('click', nextQuestion);
        restartButton.addEventListener('click', restartQuiz);

        // Mulai kuis saat halaman dimuat
        showQuestion();
    </script>
